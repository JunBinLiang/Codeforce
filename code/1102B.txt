class Sol{
    public void solution(PrintWriter out,int A[],int k){
        if(A.length<k){
            out.println("NO");
            return;
        }
        Map<Integer,List<Integer>>f=new HashMap<>();
        for(int i=0;i<A.length;i++){
            if(!f.containsKey(A[i]))f.put(A[i],new ArrayList<>());
            f.get(A[i]).add(i);
        }

        for(Integer key:f.keySet()){
            if(f.get(key).size()>k){
                out.println("NO");
                return;
            }
        }

        int res[]=new int[A.length];
        int ith=0;
        for(Integer key:f.keySet()){
            List<Integer>list=f.get(key);
            for(int index:list){
                res[index]=ith+1;
                ith++;
                ith%=k;
            }
        }

        out.println("YES");
        for(int i:res){
            out.print(i+" ");
        }
    }








}
