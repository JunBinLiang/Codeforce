class Solution {
    PrintWriter out;
    int INF = Integer.MAX_VALUE;
    int MOD = 1000000007;
    long mod = 1000000007;

    public Solution(PrintWriter out) {
        this.out = out;
    }



    public void solution(long n,long sum,long d,long w) {

        for(long i=0;i<=2*d;i++){
            long remain=sum-i*w;
            if(remain<0)break;

            if(remain%d==0){
                long x = i;
                long y = remain/d;
                if(x+y<=n){
                    out.println(y+" "+x+" "+(n-x-y));
                    return;
                }
            }
        }
        out.println(-1);
    }



}
