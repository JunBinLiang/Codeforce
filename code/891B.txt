class Solution {
    PrintWriter out;
    int INF = Integer.MAX_VALUE;
    int MOD = 1000000007;
    long mod = 1000000007;

    public Solution(PrintWriter out) {
        this.out = out;
    }

    public void solution(int A[]) {
        int B[]=new int[A.length];
        for(int i=0;i<A.length;i++){
            B[i]=A[i];
        }
        Arrays.sort(B);
        Map<Integer,Integer>f=new HashMap<>();
        for(int i=0;i<B.length;i++)f.put(B[i],i);

        for(int i=0;i<A.length;i++){
            int cur = A[i];
            int index=f.get(cur);
            A[i]=(B[(index+1)%A.length]);
            out.print(A[i]+" ");
        }
    }

}
