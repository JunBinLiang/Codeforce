class Solution {
    PrintWriter out;
    int INF = Integer.MAX_VALUE;
    int NINF = Integer.MIN_VALUE;
    int MOD = 998244353;
    int mod = 1000000007;
    Main.FastScanner fs;
    public Solution(PrintWriter out,Main.FastScanner fs) {
        this.out = out;
        this.fs=fs;
    }


    public void solution(int testcase) {
        int n =fs.Int();
        int in[]=new int[n];
        for(int i = 0; i < n - 1; i++){
            int u = fs.Int()-1;
            int v = fs.Int()-1;
            in[u]++;
            in[v]++;
        }

        int cnt = 0;
        for(int i:in){
            if(i==1)cnt++;
        }



        long x = cnt * pow(2,n - cnt) * 2;
        x %= mod;
        long y = (n - cnt) * pow(2, n - cnt);
        y %= mod;
        out.println((x+y)%mod);

    }

    public long pow(long a,long b){
        long res=1;
        long pow=a;
        for(int i=0;i<32;i++){
            if((b&(1<<i))!=0){
                res*=pow;
                res%=mod;
            }
            pow*=pow;
            pow%=mod;
        }
        return res;
    }




}
