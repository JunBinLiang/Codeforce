class Solution{
    public void solution(String s){

        char A[]=s.toCharArray();
        int res=Integer.MAX_VALUE;

        char B[]=new char[6];
        for(int i=0;i<10;i++){
            for(int j=0;j<10;j++){
                for(int k=0;k<10;k++){
                    for(int x=0;x<10;x++){
                        for(int y=0;y<10;y++){
                            for(int z=0;z<10;z++){
                                B[0]=(char)(i+'0');
                                B[1]=(char)(j+'0');
                                B[2]=(char)(k+'0');
                                B[3]=(char)(x+'0');
                                B[4]=(char)(y+'0');
                                B[5]=(char)(z+'0');
                                if(check(B)){
                                    res=Math.min(res,compare(A,B));
                                }
                            }

                        }
                    }
                }
            }
        }
        System.out.println(res);

    }

    public boolean check(char A[]){
        if(A[0]+A[1]+A[2]==A[3]+A[4]+A[5])return true;
        return false;
    }

    public int compare(char A[],char B[]){
        int cnt=0;
        for(int i=0;i<A.length;i++){
            if(A[i]!=B[i]){
                cnt++;
            }
        }
        return cnt;
    }



}
