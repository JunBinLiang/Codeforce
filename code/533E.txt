class Solution {
    PrintWriter out;
    int INF = Integer.MAX_VALUE;
    int NINF = Integer.MIN_VALUE;
    int MOD = 998244353;
    int mod = 1000000007;
    Main.FastScanner fs;
    public Solution(PrintWriter out,Main.FastScanner fs) {
        this.out = out;
        this.fs=fs;
    }




    public void solution(int testcase) {
        int n =fs.Int();
        String s = fs.Str();
        String t = fs.Str();
        char A[]=s.toCharArray();
        char B[]=t.toCharArray();
        int l = -1, r = n;
        for(int i = 0; i < A.length;i++){
            if(A[i]!=B[i]){
                l=i;
                break;
            }
        }

        for(int i = n - 1; i >= 0; i--){
            if(A[i]!=B[i]){
                r=i;
                break;
            }
        }

        int res = 0;
        if(s.substring(l,r).equals(t.substring(l+1,r+1))){
            res++;
        }
        if(s.substring(l+1,r+1).equals(t.substring(l,r))){
            res++;
        }

        out.println(res);

    }

}
