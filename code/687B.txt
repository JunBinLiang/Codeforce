class Sol{
    public void solution(PrintWriter out,int A[],int k){
        
        long lcm=A[0]%k;
        for(int i=1;i<A.length;i++){
            long g=gcd(A[i],lcm);
            lcm=lcm*A[i]/g;
            lcm%=k;
        }


        if(lcm==0){
            out.println("YES");
        }
        else{
            out.println("NO");
        }

    }

    public long gcd(long num1, long num2) {
        if (num2 != 0){
            return gcd(num2, num1 % num2);
        } else{
            return num1;
        }
    }

}
