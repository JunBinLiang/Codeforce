class Solution {
    PrintWriter out;
    int INF = Integer.MAX_VALUE;
    int NINF = Integer.MIN_VALUE;
    int MOD = 998244353;
    int mod = 1000000007;
    Main.FastScanner fs;

    int match[];
    public Solution(PrintWriter out,Main.FastScanner fs) {
        this.out = out;
        this.fs=fs;

    }

    public void solution(int testcase) {
        int n = fs.Int();
        //(x,y)
        //(x+y,y)
        //(x,x+y)
        int res = Integer.MAX_VALUE;
        for(int i = 1; i <= n; i++){
            int x = n;
            int y = i;
            int cnt = 0;
            while(x!=1 && y != 1){
                if(x>y){
                    if(y==0){
                        cnt = 100000000;
                        break;
                    }
                    int a = x/y;
                    cnt+=a;
                    x%=y;
                }
                else{
                    if(x==0){
                        cnt = 100000000;
                        break;
                    }
                    int a = y / x;
                    cnt+=a;
                    y %= x;
                }
            }
            int mx = Math.max(x,y);
            cnt += (mx - 1);
            res = Math.min(res,cnt);
        }
        out.println(res);
    }

}
