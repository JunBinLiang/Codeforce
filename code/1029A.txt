class Solution{
    PrintWriter out;
    int INF=Integer.MAX_VALUE;
    int MOD=1000000007;
    public Solution(PrintWriter out){
        this.out=out;
    }



    public void solution(String s,int k){
        int p[]=kmp(s);
        int common=p[p.length-1];
        StringBuilder str=new StringBuilder();

        str.append(s);
        for(int i=1;i<k;i++){
            str.append(s.substring(common));
        }
        out.println(str.toString());

    }

    public int[] kmp(String pattern){
        int pre[]=new int[pattern.length()];
        int l=0,r=1;
        while(r<pattern.length()){
            if(pattern.charAt(r)==pattern.charAt(l)){
                pre[r]=l+1;
                l++;
                r++;
            }else{
                if(l==0)r++;
                else{
                    l=pre[l-1];
                }
            }
        }
        return pre;
    }
